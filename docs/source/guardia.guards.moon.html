1 --- Set of guards that come with guardia. Batteries included!
    2 -- @module guardia.guards
    3 -- @author daelvn
    4 import Guard, Antiguard, Biguard, Endguard, Errguard, Antierrguard, Enderrguard from require "guardia"
    5 id     = (...) -> ...
    6 gm, sm = getmetatable, setmetatable
    7 
    8 _isType = (ty, t=type) -> (v) ->
    9   if ("table" == t v) and (gm v).__type
   10     switch t (gm v).__type
   11       when "string"   then return ty == (gm v).__type
   12       when "function" then return ty == ((gm v).__type)!
   13       else                 return ty == t v
   14   else return ty == t v
   15 --- Type-checking errguards.
   16 -- `guardia.guards` automatically creates functions such as `getsNumber` or `getsNil` for all Lua base types.
   17 -- It also supports the `__type` metafield for custom types.
   18 -- @tparam string ty Type to check for.
   19 -- @tparam function|nil t Type function to use.
   20 -- @treturn function Function that takes another to guard.
   21 getsType = (ty, t=type) -> (Errguard _isType ty, t) id
   22 
   23 _baseTypes  = {"number", "string", "function", "thread", "userdata", "nil", "boolean", "table"}
   24 _capitalize = (v) -> (string.upper string.sub v, 1, 1) .. string.sub v, 2
   25 _typeGuards = {("gets".._capitalize v), (getsType v) for v in *_baseTypes}
   26 import
   27   getsNumber,
   28   getsString,
   29   getsFunction,
   30   getsThread
   31   getsUserdata,
   32   getsNil,
   33   getsBoolean,
   34   getsTable from _typeGuards
   35 
   36 --- Type-checking enderrguards.
   37 -- `guardia.guards` automatically creates functions such as `expectsNumber` or `expectsNil` for all Lua base types.
   38 -- @tparam string ty Type to check for.
   39 -- @tparam function|nil t Type function to use.
   40 -- @treturn function Function that takes another to guard.
   41 expectsType = (ty, t=type) -> (Enderrguard _isType ty, t) id
   42 
   43 _typeEndguards = {("expects".._capitalize v), (expectsType v) for v in *_baseTypes}
   44 import
   45   expectsNumber,
   46   expectsString,
   47   expectsFunction,
   48   expectsThread
   49   expectsUserdata,
   50   expectsNil,
   51   expectsBoolean,
   52   expectsTable from _typeGuards
   53 --- Type-checking antierrguards.
   54 -- `guardia.guards` automatically creates functions such as `exceptNumber` or `exceptNil` for all Lua base types.
   55 -- @tparam string ty Type to check for.
   56 -- @tparam function|nil t Type function to use.
   57 -- @treturn function Function that takes another to guard.
   58 exceptType = (ty, t=type) -> (Antierrguard _isType ty, t) id
   59 
   60 __typeAntiguards = {("except".._capitalize v), (exceptType v) for v in *_baseTypes}
   61 import
   62   exceptNumber,
   63   exceptString,
   64   exceptFunction,
   65   exceptThread
   66   exceptUserdata,
   67   exceptNil,
   68   exceptBoolean,
   69   exceptTable from _typeGuards
   70 
   71 --- Forcefully getting a string (using `tostring`). \[Guard\].
   72 -- @treturn function Function that expects another to guard.
   73 getsStringForced = -> (Guard ->true) (v) -> tostring v
   74 
   75 --- Checks whether a number passed is odd.
   76 -- @treturn function Function that expects another to guard.
   77 getsOdd = -> (Guard (n)->n%2==0) id
   78 
   79 --- Checks whether a number passed is even.
   80 -- @treturn function Function that expects another to guard.
   81 getsEven = -> (Antiguard (n)->n%2==0) id
