1 local unpack = unpack or table.unpack
    2 local Guard
    3 Guard = function(filter)
    4   if filter == nil then
    5     filter = function()
    6       return true
    7     end
    8   end
    9   return function(transform)
   10     return function(f)
   11       return function(...)
   12         if not (filter(...)) then
   13           return ...
   14         end
   15         local argl = {
   16           transform(...)
   17         }
   18         return f(unpack(argl))
   19       end
   20     end
   21   end
   22 end
