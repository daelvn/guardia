1 --> # guardia.v2.init
    2 --> Guards for Lua. Based upon [guard.lua](https://github.com/Yonaba/guard.lua).
    3 --> This version is rather thought for curried functions and use with MoonScript.
    4 inspect_count = 0
    5 
    6 --> ## Sources
    7 --> Sources are functions that take a set of values and start a guard chain. They're always
    8 --> of the kind `(...) -> boolean, ...`.
    9 
   10 --> ### _filter
   11 --# _fl
   12 --: _filter (function) -> (...) -> boolean, ...
   13 --> Alias → `_fl`<br>
   14 --> Filtering function. If the filter passes, the first returned value is `true`, `false` otherwise.
   15 _filter = (fl) -> (...) -> if fl ... then true, ... else false, ...
   16 
   17 --> ### _true
   18 --# _t
   19 --: _true (...) -> true, ...
   20 --> Alias → `_t`<br>
   21 --> Starts a chain, always returning `true`.
   22 _true = (...) -> true, ...
   23 
   24 --> ### _false
   25 --# _f
   26 --: _false (...) -> false, ...
   27 --> Alias → `_f`<br>
   28 --> Starts a chain, always returning `false`.
   29 _false = (...) -> false, ...
   30 
   31 --> ### _default
   32 --# _df
   33 --: _default (a) -> (...) -> false, ...
   34 --> Alias → `_df`<br>
   35 --> Makes the value of the chain `a` only if `...` is `nil`.
   36 _default = (v) -> (...) -> return (_transform ->v) (_filter (x)->"nil"==type x) ...
   37 
   38 --> ## Transformers
   39 --> Transformers are functions which continue a chain, also modifying the arguments that go along
   40 --> with the boolean. They're always of the kind `(boolean, ...) -> boolean, ...`.
   41 
   42 --> ### _transform
   43 --# _tr
   44 --: _transform (function) -> (boolean, ...) -> boolean, ...
   45 --> Alias → `_tr`<br>
   46 --> Transforms the values being passed through a function, only if the boolean passed is `true`.
   47 _transform = (tr) -> (b, ...) -> if b then true, tr ... else false, ...
   48 
   49 --> ### _negate
   50 --# _ng
   51 --: _negate (boolean, ...) -> boolean, ...
   52 --> Alias → `_ng`<br>
   53 --> Negates the status from last operation.
   54 _negate = (b, ...) -> not b, ...
   55 
   56 --> ### _error
   57 --# _er
   58 --: _error (msg:string) -> (boolean, ...) -> false, ...
   59 --> Alias → `_er`<br>
   60 --> Raises → Error message passed to the function (if boolean is `false`).<br>
   61 --> Errors with `msg` if the status is `true`.
   62 _error = (msg) -> (b, ...) -> if b then error msg else false, ...
   63 
   64 --> ### _e1
   65 --# _e1
   66 --: _e1 (boolean, ...) -> false, ...
   67 --> Raises → "guardia $ chain was stopped"<br>
   68 --> Predefined error for [_error](#_error).
   69 _e1 = _error "guardia $ chain was stopped"
   70 
   71 --> ### _inspect
   72 --# _in
   73 --: _inspect (boolean, ...) -> boolean, ...
   74 --> Alias → `_in`<br>
   75 --> Prints an inspect count number and every value on the chain, including the boolean.
   76 _inspect = (b, ...) ->
   77   inspect_count += 1
   78   print inspect_count, b, ...
   79   return b, ...
   80 
   81 --> ### _named
   82 --# _ci
   83 --: _named (msg:string) -> (boolean, ...) -> boolean, ...
   84 --> Alias → `_ci`<br>
   85 --> Complex inspect. Instead of the inspect count number, prints the message along with every value on the
   86 --> chain and the status.
   87 _named = (msg) -> (b, ...) ->
   88   print msg, b, ...
   89   return b, ...
   90 
   91 --> ## Finalizers
   92 --> Finalizers are functions that end a guard chain, returning only part of the values. Their form
   93 --> can either be `(boolean, ...) -> boolean` or `(boolean, ...) -> ...`.
   94 
   95 --> ### _pass
   96 --# _ps
   97 --: _pass (boolean, ...) -> ...
   98 --> Alias → `_ps`, `_fn`<br>
   99 --> Ignores the boolean and returns the varargs.
  100 _pass = (b, ...) -> ...
  101 
  102 --> ### _status
  103 --# _st
  104 --: _status (boolean, ...) -> boolean
  105 --> Alias → `_st`<br>
  106 --> Returns the status from the chain and nothing else.
  107 _status = (b, ...) -> b
  108 
  109 
  110 _finalize = _pass
  111 _fn       = _finalize
  112 _ps       = _pass
  113 _fl       = _filter
  114 _tr       = _transform
  115 _ng       = _negate
  116 _er       = _error
  117 _st       = _status
  118 _in       = _inspect
  119 _ci       = _named
  120 _df       = _default
  121 _t        = _true
  122 _f        = _false

  125 --> ## v1 Guards
  126 --> These guards are v1 replicas made using the new guard functions. They are the closest to
  127 --> the original `guard.lua`. Most of them take the form `(function) -> (function) -> (function) -> (...) -> ...`.
  128 --- guard.lua-style Guard. **Curried function.**
  129 
  130 --> ### Guard
  131 --# Guard
  132 --: Guard (fl:function) -> (tr:function) -> (f:function) -> (...) -> ...
  133 --> Uses a filter and transformer function on a set of values passed to `f`.
  134 Guard        = (fl) -> (tr)     -> (f) -> (...) -> f _fn (_tr tr) (_fl fl) ...
  135 
  136 --> ### Antiguard
  137 --# Antiguard
  138 --: Antiguard (fl:function) -> (tr:function) -> (f:function) -> (...) -> ...
  139 --> Equivalent to [Guard](#Guard), but negates the results of `fl`.
  140 Antiguard    = (fl) -> (tr)     -> (f) -> (...) -> f _fn (_tr tr) _ng (_fl fl) ...
  141 
  142 --> ### Biguard
  143 --# Biguard
  144 --: Biguard (fl:function) -> (tr:function, ar:function) -> (f:function) -> (...) -> ...
  145 --> Essentially a mix of [Guard](#Guard) and [Antiguard](#Antiguard), so the transformer `tr` runs
  146 --> when the filter passes, `ar` otherwise.
  147 Biguard      = (fl) -> (tr, ar) -> (f) -> (...) -> f _fn (_tr ar) _ng (_fl fl) _ps (_tr tr) (_fl fl) ...
  148 
  149 --> ### Endguard
  150 --# Endguard
  151 --: Endguard (fl:function) -> (tr:function) -> (f:function) -> (...) -> ...
  152 --> A guard for the result values of the function `f` called with `...`.
  153 Endguard     = (fl) -> (tr)     -> (f) -> (...) -> _fn (_tr tr) (_fl fl) f ...
  154 
  155 --> ### Errguard
  156 --# Endguard
  157 --> [Guard](#Guard) but errors if the filter does not pass.
  158 Errguard     = (fl) -> (tr)     -> (f) -> (...) -> f _fn _e1 (_fl fl) f ...
  159 
  160 --> ### Enderrguard
  161 --# Enderrguard
  162 --> [Endguard](#Endguard) but errors if the filter does not pass.
  163 Enderrguard  = (fl) -> (tr)     -> (f) -> (...) -> _fn _e1 f ...
  164 
  165 --> ### Antierrguard
  166 --# Antierrguard
  167 --> [Antiguard](#Antiguard) but errors if the filter does not pass.
  168 Antierrguard = (fl) -> (tr)     -> (f) -> (...) -> f _fm _e1 _ng (_fl fl) f ...
